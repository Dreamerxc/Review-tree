#### 最长连续序列：

给定一个未排序的整数数组 `nums` ，找出数字连续的最长序列（不要求序列元素在原数组中连续）的长度。



```
输入：nums = [100,4,200,1,3,2]
输出：4
解释：最长数字连续序列是 [1, 2, 3, 4]。它的长度为 4。
```

##### 一，排序 + 统计  (时间复杂度o(nlogn))

```
int longestConsecutive(vector<int>& nums) {
        // 排序
        sort(nums.begin(),nums.end());
        int res = 1, m = nums.size();
        if(m==0) return 0;
        int temp = 1;
        for(int i = 1;i<m;i++){
            if(nums[i]==nums[i-1]) continue;
            if(nums[i]==nums[i-1]+1) temp++;
            else{
                res = max(temp,res);
                temp = 1;
            } 
        }
        res = max(res,temp);
        return res;
    }
```

##### 二，哈希表

```
 int longestConsecutive(vector<int>& nums) {
        int m = nums.size();
        if (m==0) return 0;
        int res = 1;
        unordered_set<int> st;
        for (auto& x:nums){
            st.insert(x);
        }
        for(int i = 0;i<m;i++) {
            if(st.find(nums[i]-1)==st.end()){
                int temp = 1;
                int val = nums[i] + 1;
                while (st.find(val)!=st.end()){
                    temp++;
                    val++;
                }
                res = max(res,temp);
            }
        }
        return res;
    }
```
