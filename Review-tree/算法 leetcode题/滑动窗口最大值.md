#### 滑动窗口最大值

##### 方法一：单调队列

```
vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int m = nums.size();
        deque<int> qu;
        vector<int> res(m-k+1);
        for (int i = 0;i<m;i++) {
            if (!qu.empty()&&i-qu.front()>=k) qu.pop_front();
            if (qu.empty()) qu.push_back(i);
            else {
                while (!qu.empty() && nums[i] > nums[qu.back()]) 
                    qu.pop_back();
                qu.push_back(i);
            }
            if (i>=k-1) res[i-k+1] = nums[qu.front()];
        }
        return res;
    }
```

##### 方法二：优先队列

```
vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int m = nums.size();
        priority_queue<pair<int,int>> qu;
        for (int i = 0;i<k-1;i++) {
            qu.emplace(nums[i],i);
        }
        vector<int> res (m-k+1);
        for (int i = k-1;i<m;i++) {
            while (!qu.empty()&&i-qu.top().second>=k) qu.pop();
            qu.emplace(nums[i],i);
            res[i-k+1] = qu.top().first;
        }
        return res;
    }
```
