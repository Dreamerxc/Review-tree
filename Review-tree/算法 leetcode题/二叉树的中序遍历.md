#### 二叉树的中序遍历

给定一个二叉树的根节点 `root` ，返回它的 **中序** 遍历。



##### 方法一：递归

```
vector<int> res;
    vector<int> inorderTraversal(TreeNode* root) {
        if(root==nullptr) return res;
        inorder(root);
        return res;
    }

    void inorder(TreeNode* root){
        if(root==nullptr) return;
        inorder(root->left);
        res.push_back(root->val);
        inorder(root->right);
    }
```



##### 方法二：非递归

```
 vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        stack<TreeNode*> st;
        while(root!=nullptr||!st.empty()){
            while(root!=nullptr){
                st.push(root);
                root = root->left;
            }
            root = st.top();
            st.pop();
            res.push_back(root->val);
            root = root->right;
        }
        return res;
    }
```
