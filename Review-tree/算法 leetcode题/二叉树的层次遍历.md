#### 二叉树的层次遍历



##### queue:

```
 vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if(root==nullptr) return res;
        queue<TreeNode*> qu;
        qu.push(root);
        while(!qu.empty()){
            vector<int> temp;
            int count = qu.size();
            for(int i = 0;i<count;i++){
                auto x = qu.front();
                qu.pop();
                temp.push_back(x->val);
                if(x->left) qu.push(x->left);
                if(x->right) qu.push(x->right);
            }
            res.push_back(temp);
        }
        return res;
    }
```
