#### 排序链表

给你链表的头结点 `head` ，请将其按 **升序** 排列并返回 **排序后的链表** 。



##### 归并排序：

```
ListNode* sortList(ListNode* head) {
        return spiltList(head);
    }

    ListNode* spiltList(ListNode* node) {
        if (node == nullptr||node->next==nullptr) return node;
        ListNode* fast = node;
        ListNode* low = node;
        while(fast->next!=nullptr && fast->next->next!=nullptr) {
            fast = fast->next->next;
            low = low->next;
        }
        fast = low->next;
        low->next = nullptr;
        low = node;
        return mergeLists(spiltList(fast), spiltList(low));
    }

    ListNode* mergeLists(ListNode* l1, ListNode* l2) {
        ListNode* head = new ListNode(-1);
        ListNode* node = head;
        while(l1!=nullptr&&l2!=nullptr) {
            if(l1->val < l2->val) {
                node->next = l1;
                l1 = l1->next; 
            }
            else {
                node->next = l2;
                l2 = l2->next;
            }
            node = node->next;
        }
        if(l1==nullptr) node->next = l2;
        else node->next = l1;
        return head->next;
    }
```
