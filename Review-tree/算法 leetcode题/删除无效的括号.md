#### 删除无效的括号 301

给你一个由若干括号和字母组成的字符串 `s` ，删除最小数量的无效括号，使得输入的字符串有效。

返回所有可能的结果。答案可以按 **任意顺序** 返回。



```
输入：s = "()())()"
输出：["(())()","()()()"]
```



##### 方法一：dfs

```
set<string> res;
    int count = 0;
    vector<string> removeInvalidParentheses(string s) {
        int m = s.size(), sum = 0;
        if (m == 0) return {};
        string temp = "";
        dfs(s,0,sum,temp);
        vector<string> ret(res.begin(), res.end());
        return ret;
    }

    void dfs(string& s, int index, int& sum, string& temp) {
        if (sum < 0) return;
        if (sum == 0) {
             if (temp.size() > count) {
                res.clear();
                res.insert(temp);
                count = temp.size();
            }
            else if (temp.size() == count) res.insert(temp);
        }
        for (int i = index; i < s.size(); i++) {
            int t = sum;
            if (s[i] == ')') sum--;
            if (s[i] == '(') sum++;
            temp.push_back(s[i]);
            dfs(s, i+1, sum, temp);
            temp.pop_back();
            sum = t;
        }
    }
```
