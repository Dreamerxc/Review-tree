#### 多数元素

给定一个大小为 n 的数组，找到其中的多数元素。多数元素是指在数组中出现次数 大于 ⌊ n/2 ⌋ 的元素。

你可以假设数组是非空的，并且给定的数组总是存在多数元素。



##### 方法一：分治算法

```
int majorityElement(vector<int>& nums) {
        // 分治算法
        return majorityElement(nums,0,nums.size()-1);
    }

    int elemcount(vector<int>& nums, int val, int low, int high){
        int count = 0;
        for(int i = low;i<=high;i++) {
            if(nums[i]==val){
                count++;
            }
        }
        return count;
    }
    int majorityElement(vector<int>& nums, int low, int high) {
        if (low == high) {
            return nums[low];
        }
        int mid = low + (high - low)/2;
        int left = majorityElement(nums, low, mid);
        int right = majorityElement(nums, mid+1, high);

        int leftcount = elemcount(nums, left, low, mid);
        int rightcount = elemcount(nums, right, mid+1, high);

        return leftcount>rightcount?left:right;
    }
```

##### 方法二：摩尔投票法

```
int majorityElement(vector<int>& nums) {
        int count = 1, res = nums[0];
        for(int i = 1;i<nums.size();i++) {
            if (count == 0) {
                count = 1;
                res = nums[i];
            }
            else if (nums[i] == res) {
                count++;
            }
            else {
                count--;
            }
        }
        return res;
    }
```
