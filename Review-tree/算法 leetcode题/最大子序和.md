#### 最大子序和：

给定一个整数数组 `nums` ，找到一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和

```
输入：nums = [-2,1,-3,4,-1,2,1,-5,4]
输出：6
解释：连续子数组 [4,-1,2,1] 的和最大，为 6 。
```



##### 方法一：动态规划

```
 int maxSubArray(vector<int>& nums) {
        int m = nums.size();
        int res = 0;
        if(m==0) return res;
        vector<int> dp(m,0);
        // dp[i]代表以第i个数据结尾的最大和
        dp[0] = nums[0];
        res = dp[0];
        for(int i = 1;i<m;i++){
            dp[i] = max(dp[i-1]+nums[i],nums[i]);
            res = max(res,dp[i]);
        }
        return res;
    }
```

##### 动态规划精简版：

```
 int maxSubArray(vector<int>& nums) {
        int m = nums.size();
        int res = 0;
        if(m==0) return res;
        int pre = nums[0];
        res = nums[0];
        for(int i = 1;i<m;i++){
            pre = max(pre+nums[i],nums[i]);
            res = max(res,pre);
        }
        return res;
    }
```

方法二：贪心算法

```
int maxSubArray(vector<int>& nums) {
        int m = nums.size();
        int res = INT_MIN;
        int sum = 0;
        for(int i = 0;i<m;i++){
            sum += nums[i];
            res = max(res,sum);
            if(sum<0){
                sum = 0;
            }
        }
        return res;
    }
```
