#### 单词拆分

给定一个非空字符串 s 和一个包含非空单词的列表 wordDict，判定 s 是否可以被空格拆分为一个或多个在字典中出现的单词。

说明：

拆分时可以重复使用字典中的单词。
你可以假设字典中没有重复的单词。



```
输入: s = "leetcode", wordDict = ["leet", "code"]
输出: true
解释: 返回 true 因为 "leetcode" 可以被拆分成 "leet code"。
```



##### 方法一：回溯（超时）

```
 bool wordBreak(string s, vector<string>& wordDict) {
        // 回溯
        set<string> st;
        for (auto& x: wordDict){
            st.insert(x);
        }
        return backtrack(s,st,0);
    }

    bool backtrack(string s, set<string>& st, int index){
        if (index>=s.size()) return true;
        for (int i = index + 1;i<=s.size();i++){
            string x = s.substr(index,i - index);
            if (st.find(x)!=st.end()){
                if (backtrack(s,st,i)) return true;
            }
        }
        return false;
    }
```

##### 方法二：回溯 + 记忆化

```
bool wordBreak(string s, vector<string>& wordDict) {
        // 回溯 + 记忆化
        int m = s.size();
        set<string> st;
        for (auto& x: wordDict){
            st.insert(x);
        }
        map<int,bool> memo;
        return backtrack(s,st,0,memo);
    }

    bool backtrack(string s, set<string>& st, int index, map<int,bool>& memo){
        if (index==s.size()) return true;
        if (memo.find(index)!=memo.end()){
            return memo[index];
        }
        for (int i = index + 1;i<=s.size();i++){
            string x = s.substr(index,i - index);
            if (st.find(x)!=st.end()&&backtrack(s,st,i,memo)){
                memo[index] = true;
                return true;
            }
        }
        memo[index] = false;
        return false;
    }
```

##### 方法三：BFS

```
bool wordBreak(string s, vector<string>& wordDict) {
        // BFS
        queue<int> qu;
        map<string,bool> memo;
        for(auto x:wordDict){
            memo[x] = true;
        }
        int m = s.size();
        if (m==0) return true;
        qu.push(0);
        while(!qu.empty()){
            auto index = qu.front();
            qu.pop();
            for(int i = index + 1;i<=m;i++){
                string v = s.substr(index,i-index);
                if (memo[v]){
                   if (i<m){
                       qu.push(i);
                   }
                   else{
                       return true;
                   }
                }
            }
        }
        return false;
    }
```

##### 方法四：BFS + 优化

```
 bool wordBreak(string s, vector<string>& wordDict) {
        // BFS
        queue<int> qu;
        map<string,bool> memo;
        for(auto x:wordDict){
            memo[x] = true;
        }
        map<int,bool> mp;
        int m = s.size();
        if (m==0) return true;
        qu.push(0);
        while(!qu.empty()){
            auto index = qu.front();
            qu.pop();
            if(mp[index]) continue;
            mp[index] = true;
            for(int i = index + 1;i<=m;i++){
                string v = s.substr(index,i-index);
                if (memo[v]){
                   if (i<m){
                       qu.push(i);
                   }
                   else{
                       return true;
                   }
                }
            }
        }
        return false;
    }
```

##### 方法五：动态规划

```
bool wordBreak(string s, vector<string>& wordDict) {
        // 动态规划
        map<string,bool> mp;
        for(auto x:wordDict){
            mp[x] = true;
        }
        int m = s.size();
        vector<int> dp(m+1,false);
        dp[0] = true;
        for(int i = 1;i<=m;i++){
            for(int j = 0;j<i;j++){
                string x = s.substr(j,i-j);
                if(dp[j]&&mp[x]){
                    dp[i] = true;
                    break;
                }
            }
        }
        return dp[m];
    }
```
