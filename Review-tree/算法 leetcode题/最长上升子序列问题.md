#### 最长上升子序列问题  300

```
 int lengthOfLIS(vector<int>& nums) {
      int m = nums.size(), res = 0;
      vector<int> dp(m,1);
      for (int i = 0;i<m;i++) {
          for (int j = i-1;j>=0;j--) {
              if (nums[j] < nums[i])
              dp[i] = max(dp[i],dp[j]+1);
          }
          res = max(dp[i],res);
      }  
      return res;
    }
```
