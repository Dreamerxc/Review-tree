#### 反转链表

##### 方法一：迭代法

```
ListNode* reverseList(ListNode* head) {
        ListNode* pre = nullptr;
        while (head != nullptr) {
           ListNode* tmp = head->next;
           head->next = pre;
           pre = head;
           head = tmp; 
        }
        return pre;
    }
```

##### 方法二：递归

```
ListNode* reverseList(ListNode* head) {
        if (head == nullptr || head->next==nullptr) return head;
        ListNode* p = reverseList(head->next);
        head->next->next = head;
        head->next = nullptr;
        return p;
    }
```
