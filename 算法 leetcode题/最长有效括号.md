#### 最长有效括号：

给你一个只包含 `'('` 和 `')'` 的字符串，找出最长有效（格式正确且连续）括号子串的长度。

```
输入：s = "(()"
输出：2
解释：最长有效括号子串是 "()"
```



##### 方法一：将有效括号的位置置为有效，计算连续有效的最长长度

```
 int longestValidParentheses(string s) {
        int m =  s.size();
        cout<<m;
        vector<int> temp(m,1);
        stack<int> st;
        for(int i = 0;i<m;i++){
            if(st.empty()){
                st.push(i);
                continue;
            }
            if(s[i]==')'&&s[st.top()]=='('){
                    temp[i] = 0;
                    temp[st.top()] = 0;
                    st.pop();
            }else{
                st.push(i);
            }
        }
        int res = 0,x = 0;
        for(int i = 0;i<m;i++){
            if(temp[i]==0) x++;
            else{
                res = max(res,x);
                x = 0;
            } 
        }
        res = max(res,x);
        return res;
    }
```

##### 方法二：动态规划

```
 int longestValidParentheses(string s) {
        int n = s.size(),res = 0;
        vector<int> dp(n,0);
        for(int i = 1;i<n;i++){
            if(s[i]==')'&&s[i-1]=='(') dp[i] = (i>=2?dp[i-2]:0) + 2;
            if(s[i]==')'&&s[i-1]==')'){
                if(i-dp[i-1]>=1&&s[i-dp[i-1]-1]=='('){
                    dp[i] = dp[i-1] + (i-dp[i-1]-2>=0?dp[i-dp[i-1]-2]:0) + 2;
                }
            }
            res = max(res,dp[i]);
        }
        return res;
    }
```
