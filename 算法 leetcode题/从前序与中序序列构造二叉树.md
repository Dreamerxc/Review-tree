#### 从前序与中序序列构造二叉树

```
前序遍历 preorder = [3,9,20,15,7]
中序遍历 inorder = [9,3,15,20,7]
```

```
  返回：
  3
   / \
  9  20
    /  \
   15   7
```



##### 解法一：递归

```
TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int m = preorder.size();
        return buildTree(preorder,inorder,0,m-1,0,m-1);
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder, int left_pre, int right_pre, int left_in,int right_in){
        if(left_pre>right_pre) return nullptr;
        int val = preorder[left_pre];
        TreeNode* root = new TreeNode(val);
        int index;
        for(int i = left_in;i<=right_in;i++){
            if(inorder[i]==val){
                index = i;
                break;
            }
        }
        int count = index - left_in;
        root->left = buildTree(preorder,inorder,left_pre+1,left_pre+count,left_in,index-1);
        root->right = buildTree(preorder,inorder,left_pre+1+count,right_pre,index+1,right_in);
        return root;
    }
```
