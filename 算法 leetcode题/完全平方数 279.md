##### 一，BFS

```
int numSquares(int n) {
        vector<int> temp;
        for (int i = 1;i*i<=n;i++) {
            temp.push_back(i*i);
        }
        queue<int> qu;
        set<int> vist;
        qu.push(n);
        int level = 0;
        while (!qu.empty()) {
           int m = qu.size();
           level++;
           for (int j = 0;j<m;j++) {
               auto x = qu.front();
               qu.pop();
               for (int i = 0;i < temp.size();i++) {
                    int val = x - temp[i];
                    if (val < 0) {
                        break;
                    }
                    else if (val == 0) return level;
                    else if (vist.find(val) == vist.end()) {
                        qu.push(val);
                        vist.insert(val);
                    }
                }
            } 
        }
        return -1;
    }
```

##### 二，动态规划

```
int numSquares(int n) {
        vector<int> temp;
        for (int i = 1;i*i<=n;i++) {
            temp.push_back(i*i);
        }
        vector<int> dp(n+1);
        for (int i = 1;i<=n;i++) {
            dp[i] = i;
            for (int j = 0;j<temp.size();j++) {
                if (i < temp[j]) break;
                else dp[i] = min(dp[i],dp[i-temp[j]] +1);
            }
        }
        return dp[n];
    }
```
