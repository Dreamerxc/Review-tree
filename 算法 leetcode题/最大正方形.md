#### 最大正方形

![img](https://assets.leetcode.com/uploads/2020/11/26/max1grid.jpg)

##### 动态规划：

```
int maximalSquare(vector<vector<char>>& matrix) {
        int res = 0;
        int m = matrix.size();
        if (m==0) return 0;
        int n = matrix[0].size();
        vector<vector<int>> dp(m,vector<int>(n,0));
        for(int i = 0;i<n;i++) {
            if (matrix[0][i]=='1') dp[0][i] = 1;
        }
        for (int i = 0;i<m;i++) {
            if (matrix[i][0]=='1') 
            {
                res = 1;
                dp[i][0] = 1;
            }
        for (int i = 1;i<m;i++) {
            for (int j = 1;j<n;j++) {
                if (matrix[i][j] == '1') {
                    dp[i][j] = min(min(dp[i-1][j],dp[i][j-1]),dp[i-1][j-1]) + 1;
                }
                else dp[i][j] = 0;
                res = max(res,dp[i][j]);
            }
        }
        return res;
    }
```
