#### 零钱兑换

给定不同面额的硬币 coins 和一个总金额 amount。编写一个函数来计算可以凑成总金额所需的最少的硬币个数。如果没有任何一种硬币组合能组成总金额，返回 -1。

你可以认为每种硬币的数量是无限的。

```
输入：coins = [1, 2, 5], amount = 11
输出：3 
解释：11 = 5 + 5 + 1
```



##### 方法一：dfs (超时)

```
  int res = INT_MAX;
    int coinChange(vector<int>& coins, int amount) {
        // dfs
        sort(coins.begin(), coins.end());
        dfs(coins,amount,0);
        return res==INT_MAX?-1:res;
    }

    void dfs(vector<int>& coins, int amount, int count) {
        if (amount < 0) return;
        if (amount == 0)  {
            res = min(res,count);
            return;
        }
        for (int i = coins.size() - 1;i>=0;i--) {
            dfs(coins, amount-coins[i], count+1);
        }
    }
```



##### 方法二：bfs （超时）

```
int coinChange(vector<int>& coins, int amount) {
        // bfs
        sort(coins.begin(), coins.end());
        vector<bool> used(amount+1,false);
        queue<int> qu;
        qu.push(amount);
        int res = 0;
        while (!qu.empty()) {
            int m = qu.size();
            for (int i = 0;i<m;i++) {
                auto x = qu.front();
                used[x] = true;
                qu.pop();
                if (x == 0) return res;
                else {
                    for (int j = 0;j < coins.size();j++) {
                        if (x - coins[j] >= 0 && !used[x-coins[j]]) {
                            qu.push(x-coins[j]);
                        }
                        else break;
                    }
                }
            }
            res++;
        }
        return -1;
    }
```

##### 方法三：动态规划

```
int coinChange(vector<int>& coins, int amount) {
        // 动态规划
        sort(coins.begin(), coins.end());
        vector<int> dp(amount+1,10e4);
        dp[0] = 0;
        for (int i = 1;i<=amount;i++) {
            for (int j = 0;j<coins.size();j++) {
                if (i >= coins[j]) dp[i] = min(dp[i],dp[i-coins[j]]+1);
                else break;
            }
        }
        return dp[amount]==10e4?-1:dp[amount];
    }
```
